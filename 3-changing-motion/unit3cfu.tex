\question 
A Porsche 911 goes from 0 to \SI{60}{mph} (\SI{26.8}{m/s}) in \SI{2.5}{s}. What is the car's average acceleration? 

\begin{choices}
    \correctchoice \SI{10.7}{m/s^2}
    \choice \SI{0.093}{m/s^2}
    \choice \SI{24.0}{m/s^2}
    \choice \SI{0.042}{m/s^2}
\end{choices}

\question

    \choice \SI{0.10}{m/s^2}
    \correctchoice \SI{9.8}{m/s^2}
    \choice \SI{4.9}{m/s^2}
