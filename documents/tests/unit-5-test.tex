\question[5]
Mia is pulling up a \SI{20}{kg} bucket of water with an acceleration of \SI{1}{m/s^2} to quench her thirst. 

\begin{center}
\begin{tikzpicture}
    \node at (0,0) {\twemoji[height=1cm]{bucket}};
    \draw[thick] (0,0.45) -- ++(0,1.5);
\end{tikzpicture}
\end{center}

\begin{parts}
\part Using the dot below, draw a free body diagram with all the forces acting on the bucket.

\begin{center}
\begin{tikzpicture}
    \ifprintanswers
        \color{red}
    \else
        \color{white}
    \fi
    \draw[thick,->] (0,0) -- (0,1.2) node[above] {$F_\mathrm{T}$};
    \draw[thick,->] (0,0) -- (0,-0.8) node[below] {$F_g$};
    \fill[black] (0,0) circle (3pt);
\end{tikzpicture}
\end{center}

\part Write the $F_\mathrm{net}$ equation for the bucket.
\vspace{-1ex}

{\Large
\begin{equation*}
    F_{\mathrm{net},y} = 
    {\ifprintanswers \color{red} F_\mathrm{T} - F_g = m a_y \fi}
\end{equation*}
}
\vspace{1ex}

\part Solve for the tension in the cable as the bucket is being pulled up.

\begin{solutionorbox}[7cm]
We know

\begin{equation*}
    F_{\mathrm{net},y} = F_\mathrm{T} - F_g = ma_y
\end{equation*}

Adding weight $F_g$ to both sides results in

\begin{align*}
    F_\mathrm{T} &= ma_y + F_g \\[1ex]
    &= ma_y + mg \\[1ex]
    &= (\SI{20}{kg})(\SI{1}{m/s^2}) + (\SI{20}{kg})(\SI{10}{N/kg}) \\[1ex]
    &= \boxed{\SI{220}{N}}
\end{align*}
\end{solutionorbox}
\end{parts}

    \begin{solutionorbox}[13cm]
    Let $F_0 = \SI{72}{N}$ be the original force of gravity on the rocket, and $d_0$ be the distance corresponding to the rocket's original position. 
    %Let $F$ be the new gravitational force at a different position, and $r$ be the new position. 

    Newton's universal law of gravitation states
    
    \begin{equation*}
        F_g = \frac{G m_1 m_2}{d^2}
    \end{equation*}
    
    and shows that gravitational force is inversely proportional to the distance squared:
    
    \begin{equation*}
        F_g \propto \frac{1}{d^2}
    \end{equation*}
    
    Therefore, the new force $F$ with new distance $d$ as a proportion of the original force is given by
    
    \begin{equation} \label{Otve2}
        F = \frac{F_0}{\left(d/d_0\right)^2}
    \end{equation}

    At Position P, the original distance is $d_0 = \sqrt{4^2 + 3^2} = 5$ units, as shown below.
    
    \begin{center}
    \begin{tikzpicture}[x=6mm,y=6mm]

        \node[opacity=0.7] at (0,0) {\twemoji[width=8mm]{globe showing Americas}};
        \fill (-4,3) circle (3pt) node[above right] {P};
    \end{tikzpicture}
    \end{center}
    
    At Position Q, the new distance is 3 units. The ratio of the old distance so the ratio is
    
    \begin{equation*}
        \frac{d}{d_0} = \frac{3}{\sqrt{4^2 + 3^2}} = \frac{3}{5}
    \end{equation*}
    
    Therefore, by Equation \eqref{Otve2}, the gravitational force at Position Q is
    
    \begin{align*}
        F_\mathrm{Q} &= \frac{F_0}{\left(d/d_0\right)^2} \\[1ex]
        &= \frac{F_0}{(3/5)^2} \\[1ex]
        &= \frac{\SI{72}{N}}{0.36} \\[1ex]
        &= \boxed{\SI{200}{N}}
    \end{align*}
    \end{solutionorbox}
